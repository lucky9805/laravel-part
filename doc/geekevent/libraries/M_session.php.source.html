 124      /**
 125       * Fetch the current session data if it exists
 126       *
 127       * @access    public
 128       * @return    bool
 129       */
 130  	function sess_read()
 131      {
 132          // Fetch the cookie
 133          $session = $this->CI->input->cookie($this->sess_cookie_name);
 134  
 135          // No cookie?  Goodbye cruel world!...
 136          if ($session === FALSE)
 137          {
 138              log_message('debug', 'A session cookie was not found.');
 139              return FALSE;
 140          }
 141  
 142          // Decrypt the cookie data
 143          if ($this->sess_encrypt_cookie == TRUE)
 144          {
 145              $session = $this->CI->encrypt->decode($session);
 146          }
 147          else
 148          {
 149              // encryption was not used, so we need to check the md5 hash
 150              $hash     = substr($session, strlen($session)-32); // get last 32 chars
 151              $session = substr($session, 0, strlen($session)-32);
 152  
 153              // Does the md5 hash match?  This is to prevent manipulation of session data in userspace
 154              if ($hash !==  md5($session.$this->encryption_key))
 155              {
 156                  log_message('error', 'The session cookie data did not match what was expected. This could be a possible hacking attempt.');
 157                  $this->sess_destroy();
 158                  return FALSE;
 159              }
 160          }
 161  
 162          // Unserialize the session array
 163          $session = $this->_unserialize($session);
 164  
 165          // Is the session data we unserialized an array with the correct format?
 166          if ( ! is_array($session) OR ! isset($session['session_id']) OR ! isset($session['ip_address']) OR ! isset($session['user_agent']) OR ! isset($session['last_activity']))
 167          {
 168              $this->sess_destroy();
 169              return FALSE;
 170          }
 171  
 172          // Is the session current?
 173          if (($session['last_activity'] + $this->sess_expiration) < $this->now)
 174          {
 175              $this->sess_destroy();
 176              return FALSE;
 177          }
 178  
 179          // Does the IP Match?
 180          if ($this->sess_match_ip == TRUE AND $session['ip_address'] != $this->CI->input->ip_address())
 181          {
 182              $this->sess_destroy();
 183              return FALSE;
 184          }
 185  
 186          // Does the User Agent Match?
 187          if ($this->sess_match_useragent == TRUE AND trim($session['user_agent']) != trim(substr($this->CI->input->user_agent(), 0, 50)))
 188          {
 189              $this->sess_destroy();
 190              return FALSE;
 191          }
 192  
 193          // Is there a corresponding session in the Memcached?
 194          if ($this->sess_use_memcached === TRUE)
 195          {
 196              $session = $this->CI->memcached->get($session['session_id']);
 197              if(!$session)
 198              {
 199                  $this->sess_destroy();
 200                  return FALSE;
 201              }
 202          }
 203  
 204          // Session is valid!
 205          $this->userdata = $session;
 206          unset($session);
 207  
 208          return TRUE;
 209      }

 233  
 234          // 设置存储cookie的数组
 235          $cookie_userdata = array();
 236  
 237          // Before continuing, we need to determine if there is any custom data to deal with.
 238          // Let's determine this by removing the default indexes to see if there's anything left in the array
 239          // and set the session data while we're at it
 240          foreach (array('session_id','ip_address','user_agent','last_activity') as $val)
 241          {
 242              $cookie_userdata[$val] = $this->userdata[$val];
 243          }
 244

 271          $this->userdata = array(

 273                              'ip_address'    => $this->CI->input->ip_address(),
 274                              'user_agent'    => substr($this->CI->input->user_agent(), 0, 50),
 275                              'last_activity'    => $this->now
 276                              );
 277  
 278  
 279          // Save the data to the Memcached if needed
 280          if ($this->sess_use_memcached === TRUE)
 281          {
 282              $this->CI->memcached->add($session_id,$this->userdata,$this->sess_expiration);
 283          }
 284  
 285          // Write the cookie
 286          $this->_set_cookie();
