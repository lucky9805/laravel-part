 505     /**
 506       * Render the master template or a single region
 507       *
 508       * @access    public
 509       * @param    string    optionally opt to render a specific region
 510       * @param    boolean    FALSE to output the rendered template, TRUE to return as a string. Always TRUE when $region is supplied
 511       * @return    void or string (result of template build)
 512       */
 513     
 514     function render($region = NULL, $buffer = FALSE, $parse = FALSE)
 515     {
 516        // Just render $region if supplied
 517        if ($region) // Display a specific regions contents
 518        {
 519           if (isset($this->regions[$region]))
 520           {
 521              $output = $this->_build_content($this->regions[$region]);
 522           }
 523           else
 524           {
 525              show_error("Cannot render the '{$region}' region. The region is undefined.");
 526           }
 527        }
 528        
 529        // Build the output array
 530        else
 531        {
 532           foreach ($this->regions as $name => $region)
 533           {
 534              $this->output[$name] = $this->_build_content($region);
 535           }
 536           
 537           if ($this->parse_template === TRUE or $parse === TRUE)
 538           {
 539              // Use provided parser class and method to render the template
 540              $output = $this->CI->{$this->parser}->{$this->parser_method}($this->master, $this->output, TRUE);
 541              
 542              // Parsers never handle output, but we need to mimick it in this case
 543              if ($buffer === FALSE)
 544              {
 545                 $this->CI->output->set_output($output);
 546              }
 547           }
 548           else
 549           {
 550              // Use CI's loader class to render the template with our output array
 551              $output = $this->CI->load->view($this->master, $this->output, $buffer);
 552           }
 553        }
 554        
 555        return $output;
 556     }

 576     /**
 577       * Build a region from it's contents. Apply wrapper if provided
 578       *
 579       * @access    private
 580       * @param    string    region to build
 581       * @param    string    HTML element to wrap regions in; like '<div>'
 582       * @param    array    Multidimensional array of HTML elements to apply to $wrapper
 583       * @return    string    Output of region contents
 584       */
 585     
 586     function _build_content($region, $wrapper = NULL, $attributes = NULL)
 587     {
 588        $output = NULL;
 589        
 590        // Can't build an empty region. Exit stage left
 591        if ( ! isset($region['content']) or ! count($region['content']))
 592        {
 593           return FALSE;
 594        }
 595        
 596        // Possibly overwrite wrapper and attributes
 597        if ($wrapper)
 598        {
 599           $region['wrapper'] = $wrapper;
 600        }
 601        if ($attributes)
 602        {
 603           $region['attributes'] = $attributes;
 604        }
 605        
 606        // Open the wrapper and add attributes
 607        if (isset($region['wrapper'])) 
 608        {
 609           // This just trims off the closing angle bracket. Like '<p>' to '<p'
 610           $output .= substr($region['wrapper'], 0, strlen($region['wrapper']) - 1);
 611           
 612           // Add HTML attributes
 613           if (isset($region['attributes']) && is_array($region['attributes']))
 614           {
 615              foreach ($region['attributes'] as $name => $value)
 616              {
 617                 // We don't validate HTML attributes. Imagine someone using a custom XML template..
 618                 $output .= " $name=\"$value\"";
 619              }
 620           }
 621           
 622           $output .= ">";
 623        }
 624        
 625        // Output the content items.
 626        foreach ($region['content'] as $content)
 627        {
 628           $output .= $content;
 629        }
 630        
 631        // Close the wrapper tag
 632        if (isset($region['wrapper']))
 633        {
 634           // This just turns the wrapper into a closing tag. Like '<p>' to '</p>'
 635           $output .= str_replace('<', '</', $region['wrapper']) . "\n";
 636        }
 637        
 638        return $output;
 639     }
