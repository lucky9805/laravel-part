 239      public function execute()
 240      {
 241          // Set two default options, and merge any extra ones in
 242          if(!isset($this->options[CURLOPT_TIMEOUT]))           $this->options[CURLOPT_TIMEOUT] = 30;
 243          if(!isset($this->options[CURLOPT_RETURNTRANSFER]))    $this->options[CURLOPT_RETURNTRANSFER] = TRUE;
 244          if(!isset($this->options[CURLOPT_FOLLOWLOCATION]))    $this->options[CURLOPT_FOLLOWLOCATION] = TRUE;
 245          if(!isset($this->options[CURLOPT_FAILONERROR]))       $this->options[CURLOPT_FAILONERROR] = TRUE;
 246  
 247          if(!empty($this->headers))
 248          {
 249              $this->option(CURLOPT_HTTPHEADER, $this->headers); 
 250          }
 251  
 252          $this->options();
 253  
 254          // Execute the request & and hide all output
 255          $this->response = curl_exec($this->session);
 256  
 257          // Request failed
 258          if($this->response === FALSE)
 259          {
 260              $this->error_code = curl_errno($this->session);
 261              $this->error_string = curl_error($this->session);
 262              
 263              curl_close($this->session);
 264              $this->session = NULL;
 265              return FALSE;
 266          } 
 267          
 268          // Request successful
 269          else
 270          {
 271              $this->info = curl_getinfo($this->session);
 272              
 273              curl_close($this->session);
 274              $this->session = NULL;
 275              return $this->response;
 276          }
